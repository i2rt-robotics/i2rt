# LeRobot recorder/replay — workstation env (SEPARATE from the YAM/robot env).
#
# No ROS. The robot link is portal (plain TCP) via i2rt's RobotClient, so this
# env just needs i2rt installed (for i2rt.serving.robot_client) plus the deps
# below. The policy bridge additionally uses yam-policy (websocket client).
#
# System packages (apt) needed for LeRobot v3.0 video encoding:
#   sudo apt install -y ffmpeg
#
# Setup (uv; full guide in README.md):
#   uv venv ~/yam_ws                       # any Python >= 3.10; no ROS constraint
#   source ~/yam_ws/bin/activate
#   uv pip install -e .                     # i2rt (robot client, portal)
#   uv pip install -e policy_serving        # yam-policy (websocket client, for the bridge)
#   uv pip install -r workstation/lerobot_recorder/requirements.txt
#
# RealSense udev rules are NOT installed by the pip wheel — see README.md
# (workstation -> RealSense cameras) or cameras fail with permission errors.

lerobot>=0.4.0      # LeRobotDataset v3.0 API (create/add_frame/save_episode/finalize)
pyrealsense2        # RealSense D405 / D455 cameras
PyQt5               # GUI
numpy
